/* ЗАДАНИЕ 2
 * При помощи базы данных Redis решите задачу поиска имени пользователя по 
 * электронному адресу и наоброт, поиск электронного адреса пользователя по его имени.
 * 
 * Решение
 * Для хранения данных будем использовать 2 базы данных: 0 и 1. В первой базе будем \
 * хранить соотношение между именем пользователя и его почтой. Во второй - обратное 
 * соотношение. Для добавления пользователей нужно последовательно добавить 
 * соотношение в одну базу, затем зеркальное соотшношение в другую. Для удаления -
 * аналогично: удалить из обеих баз по одной записи. Для поиска нужно сначала переключиться
 * на нужную базу, затем задать вопрос GET
 */

127.0.0.1:6379[1]> select 0
OK
127.0.0.1:6379> set andrew and@mail.com
OK
127.0.0.1:6379> set helen hel@mail.com
OK
127.0.0.1:6379> set james jam@mail.com
OK
127.0.0.1:6379> keys *
1) "helen"
2) "james"
3) "andrew"
127.0.0.1:6379> select 1
OK
127.0.0.1:6379[1]> set and@mail.com andrew
OK
127.0.0.1:6379[1]> set hel@mail.com helen
OK
127.0.0.1:6379[1]> set jam@mail.com james
OK
127.0.0.1:6379[1]> select 0
OK
127.0.0.1:6379> get helen
"hel@mail.com"
127.0.0.1:6379> select 1
OK
127.0.0.1:6379[1]> get jam@mail.com
"james"
